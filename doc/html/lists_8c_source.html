448 sensorRecPruneAll()
449 {
450 
451  TTLS *ttlPtr = SensorTtls;
452  struct timeval now, timeDiff;
453  SENSOR_LIST *sensorListPtr;
454  SENSOR_ID_LIST *sensorIdListPtr;

459 
460  if (DebugLevel >= DEBUG_DETAIL) {
461 #ifdef __APPLE__
462  fprintf(DebugFP, "%s() enter @ %ld.%06d\n", __func__, now.tv_sec, now.tv_usec);
463 #else // ! __APPLE__
464  fprintf(DebugFP, "%s() enter @ %ld.%06ld\n", __func__, now.tv_sec, now.tv_usec);
465 #endif // __APPLE__
466  }
467 
468  LOCK_SENSOR_LIST;
469 
470  if (DebugLevel >= DEBUG_SUPER) {
471  dumpLists();
472  }
473 
474  while (ttlPtr->sensor) { // go through sensor types in the TTL list
475 
476  if (DebugLevel >= DEBUG_SUPER) {
477  fprintf(DebugFP, "%s(): pruning sensor \'%c\'\n", __func__, ttlPtr->sensor);
478  }
479 
480  sensorListPtr = sensorGetListBySensor(ttlPtr->sensor);
481 
482  sensorIdListPtr = sensorListPtr->sensors;
483 
484  while (sensorIdListPtr) { // walk the ids
485  if (DebugLevel >= DEBUG_SUPER) {
486  fprintf(DebugFP, "%s(): pruning SENSOR_ID @ (0x%lx)n",
487  __func__, (long)sensorIdListPtr);
488  }
489 
490  sensorSubIdListPtr = sensorIdListPtr->subIds;
491 
492  while (sensorSubIdListPtr) { // walk the subIds
493 
494  if (DebugLevel >= DEBUG_SUPER) {
495  fprintf(DebugFP, "%s(): pruning SENSOR_SUB_ID @ (0x%lx)n",
496  __func__, (long)sensorSubIdListPtr);
497  }
498 
499  sensorPtr = sensorSubIdListPtr->data;
500  prevSensorPtr = sensorPtr;
501 
502  while (sensorPtr) { // walk the sensor data records
503 
504  if (DebugLevel >= DEBUG_SUPER) {
505  fprintf(DebugFP, "%s(): pruning SENSOR @ (0x%lx)n",
506  __func__, (long)sensorSubIdListPtr);
507  }
508 
509  timersub(&now, &(sensorPtr->time), &timeDiff);
510 
511  if (timeDiff.tv_sec > ttlPtr->ttlSecs ||
512  (timeDiff.tv_sec == ttlPtr->ttlSecs && timeDiff.tv_usec > ttlPtr->ttlUsecs)) {
513  // start pruning from here
514  if (prevSensorPtr == sensorPtr) { // prune them all
515  sensorSubIdListPtr->data = NULL;
516  } else {
517  prevSensorPtr->next = NULL;
518  }
519 
520  freeSensorRecsFromEnd(sensorPtr);
521 
522  sensorPtr = NULL;
523 
524  } else {
525 
526  prevSensorPtr = sensorPtr;
527  sensorPtr = sensorPtr->next;
528  }
529  }
530 
531  sensorSubIdListPtr = sensorSubIdListPtr->next;
532  }
533 
534  sensorIdListPtr = sensorIdListPtr->next;
535  }
536 
537  ttlPtr++; // next sensor id on the TTL list
538  }
539 
540  UNLOCK_SENSOR_LIST;
541 
542  if (DebugLevel >= DEBUG_DETAIL) {
543  gettimeofday(&now, NULL);
544 #ifdef __APPLE__
545  fprintf(DebugFP, "%s() exit @ %ld.%06d\n", __func__, now.tv_sec, now.tv_usec);
546 #else // ! __APPLE__
547  fprintf(DebugFP, "%s() exit @ %ld.%06ld\n", __func__, now.tv_sec, now.tv_usec);
548 #endif // __APPLE__
549  }
550
